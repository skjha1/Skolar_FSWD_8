```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>em Example</title>
    <style>
        body {
            font-size: 16px; /* Base font size */
        }
        h1 {
            font-size: 2em; /* 2 times of the base font size */
        }
        p {
            font-size: 1em; /* This will be same as the base font size */
        }
        .smaller-text{
            font-size: 0.75em ;  /* 75% of the base font size */
        }

    </style>
</head>
<body>
    <h1>This is a Heading</h1>
    <p>This is a paragraph</p>
    <p class="smaller-text">This is a smaller text</p>
</body>
</html>
```
